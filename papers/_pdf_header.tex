% Font fallback chain for the unicode-heavy Schur paper (lualatex + luaotfload).
% Prose uses STIX Two Text; inline-code/diagrams use Andale Mono; any glyph missing
% from those falls back to STIX Two Math, then Menlo (which carries the modifier
% letters U+1D40 transpose and U+1D62 subscript-i that STIX Math lacks).
\directlua{
  luaotfload.add_fallback("schurfb", {
    "STIX Two Math:mode=base;",
    "Menlo:mode=base;",
  })
}
\setmainfont{STIX Two Text}[RawFeature={fallback=schurfb}]
\setmonofont{Andale Mono}[RawFeature={fallback=schurfb}, Scale=0.90]
\usepackage{microtype}
% Code blocks (the wide two-axis map is ~93 mono chars) need to fit the text column.
\usepackage{etoolbox}

\AtBeginEnvironment{Highlighting}{\footnotesize}
\AtBeginEnvironment{Shaded}{\footnotesize}
